def performance_metric2(y_true, y_predict):
    """"""
    y_true = np.array(y_true)
    y_predict = np.array(y_predict)
    y_true_average = np.average(y_true)
    ss_res = np.sum(np.square(y_true - y_predict))
    ss_tot = np.sum(np.square(y_true - y_true_average))  
    score = 1-ss_res/ss_tot
    return score

from sklearn.tree import DecisionTreeRegressor

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=10):
    """  """
    scores = []
    for i in range(cv):
        # 
        X_test = X[i*(len(X)/cv):(i+1)*(len(X)/cv)]
        y_test = y[i*(len(X)/cv):(i+1)*(len(X)/cv)]
        traning_set_indice = list(set(range(len(X)))-set(range(i*(len(X)/cv),(i+1)*(len(X)/cv)))) 
        X_train = X.loc[traning_set_indice]
        y_train = y.loc[traning_set_indice]
        
        # NaN
        '''
        X_train = X_train.dropna()
        y_train = y_train.dropna()
        X_test = X_test.dropna()
        y_test =y_test.dropna()
        '''
        X_train = X_train.fillna(0)
        y_train = y_train.fillna(0)
        X_test = X_test.fillna(0)
        y_test =y_test.fillna(0)
        
        # 
        estimator = estimator.fit(X_train, y_train)
        # 
        score = performance_metric(y_test, estimator.predict(X_test))
        scores.append(score)
    return scores


def fit_model2(X, y):
    """  [X,y]"""
    #
    max_depths =[i for i in range(1,11)]
    best_score = 0
    best_estimator = None
    for max_depth in max_depths:
        estimator = DecisionTreeRegressor(max_depth = max_depth)
        # 
        score = np.mean(cross_val_score(estimator, X, y, scoring = performance_metric, cv=10))
        if score>best_score:
            best_score = score
            best_estimator = estimator            
    return best_estimator
